106  ExecutionThread(trajectory_msgs::JointTrajectory &trajectory) {
107  actionlib::SimpleActionClient <control_msgs::FollowJointTrajectoryAction> ac("/arm_controller/follow_joint_trajectory", true);
108  ac.waitForServer();
109  control_msgs::FollowJointTrajectoryGoal trajectory_goal;
110  trajectory.header.stamp = ros::Time::now();
111  trajectory_goal.trajectory = trajectory;
112  ac.sendGoal(trajectory_goal);
113  bool finished_before_timeout = ac.waitForResult(ros::Duration(30.0));
114  if (finished_before_timeout) {
115  actionlib::SimpleClientGoalState state = ac.getState();
116  spdlog::get("file_logger")->info("[{}]: {} {}.", __FUNCTION__, "Trajectory execution finished with ", state.toString().c_str());
117  return true;
118  } else {
119  spdlog::get("file_logger")->info("[{}]: {}", __FUNCTION__, "Trajectory execution timed out");
120  return false;
121  }
122  }

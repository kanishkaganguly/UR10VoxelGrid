58  void ExecuteRobotCommand(const std::string &joint_name, const double &joint_target) {

109  void ExecuteAction(const voxel_grid_plugin::ModelControlGoalConstPtr &command) {
110  std::string pkg_path = ros::package::getPath("voxel_grid_plugin");
111  std::string models_path = pkg_path + "/models/obstacle_box_";
112  switch (command->model_command) {
113  case voxel_grid_plugin::ModelControlGoal::MODEL_ADD: {
114  spdlog::get("file_logger")->info("[{}]: {}.", __FUNCTION__, "Adding model");
115  if (!this->CheckModelExists(static_cast<std::string>(command->model_name))) {
116  spdlog::get("file_logger")->info("[{}]: {}.", __FUNCTION__, "Model does not exist in world");
117  switch (command->model_size) {
118  case voxel_grid_plugin::ModelControlGoal::MODEL_SMALL:
119  spdlog::get("file_logger")->info("[{}]: {}.", __FUNCTION__, "Small model selected");
120  models_path += "small.sdf";
121  break;
122  case voxel_grid_plugin::ModelControlGoal::MODEL_MED:
123  spdlog::get("file_logger")->info("[{}]: {}.", __FUNCTION__, "Medium model selected");
124  models_path += "med.sdf";
125  break;
126  case voxel_grid_plugin::ModelControlGoal::MODEL_LARGE:
127  spdlog::get("file_logger")->info("[{}]: {}.", __FUNCTION__, "Large model selected");
128  models_path += "large.sdf";
129  break;
130  }
131 
132  geometry_msgs::Pose pose = command->initial_pose;
133  Eigen::Quaternionf q(pose.orientation.w, pose.orientation.x, pose.orientation.y, pose.orientation.z);
134  Eigen::Vector3f euler = math_helper::eigen_conversions::EigenQuaternionToEulerXYZ(q);
135  std::stringstream ss;
136  ss << pose.position.x << " " << pose.position.y << " " << pose.position.z << " "
137  << euler(0) << " " << euler(1) << " " << euler(2);
138  std::string prefixed_name = static_cast<std::string>(voxel_grid_plugin::ModelControlGoal::MODEL_PREFIX)
139  + static_cast<std::string>(command->model_name);
140  std::string updated_name_sdf = utils.SetModelNameFromFile(models_path, command->model_name);
141  std::string updated_pose_sdf = utils.SetInitialPoseFromString(updated_name_sdf, ss.str());
142  this->world->InsertModelString(updated_pose_sdf);
143  result.finished = true;
144  } else {
145  result.finished = false;
146  }
147  break;
148  }
149  case voxel_grid_plugin::ModelControlGoal::MODEL_DELETE: {
150  spdlog::get("file_logger")->info("[{}]: {}.", __FUNCTION__, "Deleting model");
151  if (this->CheckModelExists(static_cast<std::string>(command->model_name))) {
152  this->world->RemoveModel(command->model_name);
153  result.finished = true;
154  } else {
155  result.finished = false;
156  }
157  break;
158  }
159  default:
160  result.finished = false;
161  break;
162  }
163  as.setSucceeded(result);
164  }

222  void ExecuteAction(const voxel_grid_plugin::VoxelGridGoalConstPtr &command) {
223  spdlog::get("file_logger")->info("[{}-{}] {}.", __FUNCTION__, action_name, "Received action goal");
224  // Set total number of voxels in grid
225  result.num_voxels_per_dim = voxel_grid_ptr->NUM_VOXELS_PER_DIM;
226  // Fetch latest voxel grid
227  const Tensor &grid = this->voxel_grid_ptr->GetGrid();
228  // Total voxels
229  const float total_voxels = (float) std::pow(voxel_grid_ptr->NUM_VOXELS_PER_DIM + 1, 3);
230  // Start timer
231  auto start = std::chrono::high_resolution_clock::now();
232  // Iterate through all the voxels
233  for (int x = 0; x < voxel_grid_ptr->NUM_VOXELS_PER_DIM; ++x) {
234  for (int y = 0; y < voxel_grid_ptr->NUM_VOXELS_PER_DIM; ++y) {
235  for (int z = 0; z < voxel_grid_ptr->NUM_VOXELS_PER_DIM; ++z) {
236  // Set the voxel coordinates
237  voxel_grid_plugin::VoxelData voxel_data;
238  voxel_data.voxel_coordinate.push_back(x);
239  voxel_data.voxel_coordinate.push_back(y);
240  voxel_data.voxel_coordinate.push_back(z);

248  // Add voxel data to corresponding array
249  result.occupied_nodes.push_back(voxel_data);
250  }
251  // Compute progress
252 // float progress = (((x * voxel_grid_ptr->NUM_VOXELS_PER_DIM * voxel_grid_ptr->NUM_VOXELS_PER_DIM) +
253 // (y * voxel_grid_ptr->NUM_VOXELS_PER_DIM) + z) / total_voxels) * 100;
254 // feedback.completion_percent = progress;

257  }

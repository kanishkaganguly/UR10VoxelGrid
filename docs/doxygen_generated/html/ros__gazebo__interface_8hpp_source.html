105  inline void ExecuteAction(const voxel_grid_plugin::SimControlGoalConstPtr &command) {
106  spdlog::info("[{}-{}] {}.", __FUNCTION__, action_name, "Received action goal");
107  switch (command->sim_command) {
108  case voxel_grid_plugin::SimControlGoal::SIM_INIT: {
109  spdlog::info("[{}]: {}.", __FUNCTION__, "Initializing simulator.");
110  feedback.status = "Initializing simulator.";
111  InitSim(command->sim_time_factor);
112  }
113  break;
114  case voxel_grid_plugin::SimControlGoal::SIM_STEP: {
115  spdlog::info("[{}]: {}.", __FUNCTION__, "Stepping simulator.");
116  feedback.status = "Stepping simulator.";
117  std::thread stepper_thread(&GazeboInterface::StepSim, this, command->sim_time_factor);
118  stepper_thread.join();
119  }
120  break;
121  case voxel_grid_plugin::SimControlGoal::SIM_PAUSE: {
122  spdlog::info("[{}]: {}.", __FUNCTION__, "Pausing simulator.");
123  feedback.status = "Pausing simulator.";
124  PauseSim();
125  }
126  break;
127  case voxel_grid_plugin::SimControlGoal::SIM_START: {
128  spdlog::info("[{}]: {}.", __FUNCTION__, "Pausing simulator.");
129  feedback.status = "Pausing simulator.";
130  StartSim();
131  }
132  break;
133  }
134  as.publishFeedback(feedback);
135  as.setSucceeded(result);
136  }

214  inline void ExecuteAction(const voxel_grid_plugin::RobotControlGoalConstPtr &command) {
215  spdlog::info("[{}-{}] {}.", __FUNCTION__, action_name, "Received action goal");
216  if (command->add_to_queue) {
217  spdlog::info("[{}]: {}.", __FUNCTION__, "Adding command to execution queue.");
218  feedback.status = "Adding command to execution queue";
219  AddRobotCommand(command->joint_name, command->joint_target);
220  } else if (!command->add_to_queue) {
221  spdlog::info("[{}]: {}.", __FUNCTION__, "Starting execution queue.");
222  SendRobotCommand();
223  feedback.status = "Starting execution queue";
224  }

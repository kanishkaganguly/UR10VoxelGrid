127  void ParseMapping() {
128  int iter = 0;
129  // Get root element
130  tinyxml2::XMLElement *root = this->doc.FirstChildElement("links");
131  // Loop through all children
132  for (tinyxml2::XMLElement *e = root->FirstChildElement("link"); e != NULL; e = e->NextSiblingElement("link")) {
133  // Get name attribute
134  const char *link_name_ch = e->Attribute("name");
135  std::string link_name(link_name_ch);
136  // Fetch all three child data elements (mappings)
137  tinyxml2::XMLElement *map_0 = e->FirstChildElement();
138  tinyxml2::XMLElement *map_1 = map_0->NextSiblingElement();
139  tinyxml2::XMLElement *map_2 = map_1->NextSiblingElement();
140 
141  float val = -100;
142  // Fetch float value of element
143  map_0->QueryFloatText(&val);
144  // Add data to array
145  world_link_mapping[iter] = std::make_pair(link_name, std::make_pair(0, val == 10 ? -0.0f : val));
146  // Increment array insertion index
147  iter += 1;
148 
149  map_1->QueryFloatText(&val);
150  world_link_mapping[iter] = std::make_pair(link_name, std::make_pair(1, val == 10 ? -0.0f : val));
151  iter += 1;
152 
153  map_2->QueryFloatText(&val);
154  world_link_mapping[iter] = std::make_pair(link_name, std::make_pair(2, val == 10 ? -0.0f : val));
155  iter += 1;
156  }
157  }

381  void GetCanonicalPositionsWithAxisMapping(const std::string &link_name,
382  const gazebo::math::Box &cbox,
383  std::array<Eigen::Vector4f, 4> &left_face,
384  std::array<Eigen::Vector4f, 4> &right_face) {
385  // Make vector of (axis, dimension) pairs
386  std::vector<std::pair<int, double>> box_size = {
387  std::make_pair(0, cbox.GetXLength()),
388  std::make_pair(1, cbox.GetYLength()),
389  std::make_pair(2, cbox.GetZLength())
390  };
391  // Custom comparator function for sorting axis by dimension
392  auto sortLongestAxis = [](const std::pair<int, double> &l,
393  const std::pair<int, double> &r) -> bool {
394  return l.second >= r.second;
395  };
396  std::sort(box_size.begin(), box_size.end(), sortLongestAxis);
397  std::pair<int, double> offset = box_size[0]; // Offset from box center to face center
398  std::pair<int, double> face_offset_w = box_size[1]; // Offset from face center along width
399  std::pair<int, double> face_offset_h = box_size[2]; // Offset from face center along height
400 
401  // Map world axes to corresponding link axes, based on lookup table
402  float axis_0 = GetMapping(link_name, 0); // Get axis mapping from link name
403  float axis_1 = GetMapping(link_name, 1); // Get axis mapping from link name
404  float axis_2 = GetMapping(link_name, 2); // Get axis mapping from link name
405 
406  offset.first = static_cast<int>(std::abs(axis_0)); // Replace the axis with mapped value
407  offset.second = std::signbit(axis_0) ? -offset.second : offset.second; // Replace the sign, if negative
408 
409  face_offset_w.first = static_cast<int>(std::abs(axis_1)); // Replace the axis with mapped value
410  face_offset_w.second = std::signbit(axis_1) ? -face_offset_w.second : face_offset_w.second; // Replace the sign, if negative
411 
412  face_offset_h.first = static_cast<int>(std::abs(axis_2)); // Replace the axis with mapped value
413  face_offset_h.second = std::signbit(axis_2) ? -face_offset_h.second : face_offset_h.second; // Replace the sign, if negative
414 
415  //======================== Get left face corners for canonical positions =========================//
416  // Define four corner of face in clockwise order
417  Eigen::Vector4f left_face_tl = Eigen::Vector4f::Ones(),
418  left_face_tr = Eigen::Vector4f::Ones(),
419  left_face_br = Eigen::Vector4f::Ones(),
420  left_face_bl = Eigen::Vector4f::Ones();
421  // Offset of center of box along longest axis, this gets you to center of the face
422  // Offset of center of face to get four corners
423  // Top left
424  left_face_tl(offset.first) = -offset.second / 2;
425  left_face_tl(face_offset_w.first) = face_offset_w.second / 2;
426  left_face_tl(face_offset_h.first) = -face_offset_h.second / 2;
427  // Top right
428  left_face_tr(offset.first) = -offset.second / 2;
429  left_face_tr(face_offset_w.first) = face_offset_w.second / 2;
430  left_face_tr(face_offset_h.first) = face_offset_h.second / 2;
431  // Bottom right
432  left_face_br(offset.first) = -offset.second / 2;
433  left_face_br(face_offset_w.first) = -face_offset_w.second / 2;
434  left_face_br(face_offset_h.first) = face_offset_h.second / 2;
435  // Bottom left
436  left_face_bl(offset.first) = -offset.second / 2;
437  left_face_bl(face_offset_w.first) = -face_offset_w.second / 2;
438  left_face_bl(face_offset_h.first) = -face_offset_h.second / 2;
439  // Set left face canonical points
440  left_face = {left_face_tl, left_face_tr, left_face_br, left_face_bl};
441  //======================== Get right face corners for canonical positions ========================//
442  // Define four corner of face in clockwise order
443  Eigen::Vector4f right_face_tl = Eigen::Vector4f::Ones(),
444  right_face_tr = Eigen::Vector4f::Ones(),
445  right_face_br = Eigen::Vector4f::Ones(),
446  right_face_bl = Eigen::Vector4f::Ones();
447 
448  // Offset of center of box along longest axis, this gets you to center of the face
449  // Offset of center of face to get four corners
450  // Top left
451  right_face_tl(offset.first) = offset.second / 2;
452  right_face_tl(face_offset_w.first) = face_offset_w.second / 2;
453  right_face_tl(face_offset_h.first) = -face_offset_h.second / 2;
454  // Top right
455  right_face_tr(offset.first) = offset.second / 2;
456  right_face_tr(face_offset_w.first) = face_offset_w.second / 2;
457  right_face_tr(face_offset_h.first) = face_offset_h.second / 2;
458  // Bottom right
459  right_face_br(offset.first) = offset.second / 2;
460  right_face_br(face_offset_w.first) = -face_offset_w.second / 2;
461  right_face_br(face_offset_h.first) = face_offset_h.second / 2;
462  // Bottom left
463  right_face_bl(offset.first) = offset.second / 2;
464  right_face_bl(face_offset_w.first) = -face_offset_w.second / 2;
465  right_face_bl(face_offset_h.first) = -face_offset_h.second / 2;
466  // Set right face canonical points
467  right_face = {right_face_tl, right_face_tr, right_face_br, right_face_bl};
468  };

559  void TransformCanonicalToWorld(const gazebo::math::Quaternion &rotation,
560  const gazebo::math::Vector3 &position,
561  const std::array<Eigen::Vector4f, 4> &left_canonical,
562  const std::array<Eigen::Vector4f, 4> &right_canonical,
563  std::array<Eigen::Vector4f, 4> &transformed_left_points,
564  std::array<Eigen::Vector4f, 4> &transformed_right_points) {
565  using namespace math_helper::eigen_conversions;
566  using namespace math_helper::gazebo_conversions;
567  using namespace math_helper::gazebo_eigen_conversions;
568 
569  //================ Transform from box to world =========================================================//
570  // Transformation matrix from box coordinates system to world coordinate system
571  Eigen::Matrix4f T_box_to_world = Eigen::Matrix4f::Zero();
572  Eigen::Matrix3f box_rot = GazeboQuaternionToEigenQuaternion(rotation).toRotationMatrix();
573  Eigen::Vector3f box_pos = GazeboVec3ToEigenVec3(position);
574  PoseToMatrix4(box_pos, box_rot, T_box_to_world);
575 
576  //================= Transform left face canonical points to world ======================================//
577  // Transform each canonical face corner to world
578  Eigen::Vector4f P_left_face_tl_to_world, P_left_face_tr_to_world, P_left_face_br_to_world, P_left_face_bl_to_world;
579  P_left_face_tl_to_world = T_box_to_world * left_canonical.at(0);
580  P_left_face_tr_to_world = T_box_to_world * left_canonical.at(1);
581  P_left_face_br_to_world = T_box_to_world * left_canonical.at(2);
582  P_left_face_bl_to_world = T_box_to_world * left_canonical.at(3);
583  transformed_left_points = {P_left_face_tl_to_world, P_left_face_tr_to_world, P_left_face_br_to_world, P_left_face_bl_to_world};
584 
585  //================= Transform right face points to box =================================================//
586  // Transform each canonical face corner to world
587  Eigen::Vector4f P_right_face_tl_to_world, P_right_face_tr_to_world, P_right_face_br_to_world, P_right_face_bl_to_world;

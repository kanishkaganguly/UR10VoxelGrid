196  inline void GetHitNodes(const Eigen::Vector3i &start_point, const Eigen::Vector3i &end_point,
197  std::vector<Eigen::Vector3i> &hit_nodes) {
198  spdlog::get("file_logger")->info("[{}]: {}.", __FUNCTION__, "Finding hit voxels");
199  // Include start point
200  hit_nodes.push_back(start_point);
201  // Set start and end points
202  int x1 = start_point(0);
203  int y1 = start_point(1);
204  int z1 = start_point(2);
205  int x2 = end_point(0);
206  int y2 = end_point(1);
207  int z2 = end_point(2);
208  // Absolute differences
209  int abs_dx = std::abs(x2 - x1);
210  int abs_dy = std::abs(y2 - y1);
211  int abs_dz = std::abs(z2 - z1);
212  // Step directions
213  int xs, ys, zs;
214  x2 > x1 ? xs = 1 : xs = -1;
215  y2 > y1 ? ys = 1 : ys = -1;
216  z2 > z1 ? zs = 1 : zs = -1;
217 
218  double hypotenuse = sqrt(abs_dx * abs_dx + abs_dy * abs_dy + abs_dz * abs_dz);
219  // We determine the value of t at which the ray crosses the first vertical voxel boundary and
220  // store it in variable maxX. We perform a similar computation in y and store the result in maxY. The
221  // minimum of these two values will indicate how much we can travel along the ray and still remain in the
222  // current voxel.
223  double maxX = hypotenuse * 0.5 / abs_dx;
224  double maxY = hypotenuse * 0.5 / abs_dy;
225  double maxZ = hypotenuse * 0.5 / abs_dz;
226  // We compute deltaX and deltaY. deltaX indicates how far along the ray we must move (in units of t)
227  // for the horizontal component of such a movement to equal the width of a voxel.
228  // Similarly, we store in deltaY the amount of movement along the ray
229  // which has a vertical component equal to the height of a voxel.
230  double deltaX = hypotenuse / abs_dx;
231  double deltaY = hypotenuse / abs_dy;
232  double deltaZ = hypotenuse / abs_dz;
233 
234  while (x1 != x2 || y1 != y2 || z1 != z2) {
235  if (maxX < maxY) {
236  if (maxX < maxZ) {
237  x1 = x1 + xs;
238  maxX = maxX + deltaX;
239  } else if (maxX > maxZ) {
240  z1 = z1 + zs;
241  maxZ = maxZ + deltaZ;
242  } else {
243  x1 = x1 + xs;
244  maxX = maxX + deltaX;
245  z1 = z1 + zs;
246  maxZ = maxZ + deltaZ;
247  }
248  } else if (maxX > maxY) {
249  if (maxY < maxZ) {
250  y1 = y1 + ys;
251  maxY = maxY + deltaY;
252  } else if (maxY > maxZ) {
253  z1 = z1 + zs;
254  maxZ = maxZ + deltaZ;
255  } else {
256  y1 = y1 + ys;
257  maxY = maxY + deltaY;
258  z1 = z1 + zs;
259  maxZ = maxZ + deltaZ;
260  }
261  } else {
262  if (maxY < maxZ) {
263  y1 = y1 + ys;
264  maxY = maxY + deltaY;
265  x1 = x1 + xs;
266  maxX = maxX + deltaX;
267  } else if (maxY > maxZ) {
268  z1 = z1 + zs;
269  maxZ = maxZ + deltaZ;
270  } else {
271  x1 = x1 + xs;
272  maxX = maxX + deltaX;
273  y1 = y1 + ys;
274  maxY = maxY + deltaY;
275  z1 = z1 + zs;
276  maxZ = maxZ + deltaZ;
277  }
278  }
279  hit_nodes.push_back(Eigen::Vector3i(x1, y1, z1));
280  }
281  }
